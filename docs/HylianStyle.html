font.add(family = 'Hylian', regular = 'HylianSymbols.ttf')
font.add(family = 'Triforce', regular = 'Triforce.ttf')
  1. Use it!
par(family="Triforce")
plot(c(0,1), c(0,1), type="n", xlab="X-axis", ylab="Y-axis", main="Hylian style")
points(runif(10), runif(10), pch="*")
